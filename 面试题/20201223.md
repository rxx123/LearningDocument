# 1、this的指向

```js
var length = 100;
function f1() {
    console.log(this.length);
}
f1();//调用者window
var obj = {
    length: 10,
    f2: f1
}
obj.f2();//调用者obj
var arr = [f1,2,3,4,5];
arr[0](); //调用者arr

this的指向，有没有调用者，没有调用者就是window

var length = 100;
function f1() {
    console.log(this.length)
}
var obj = {
    length: 10,
    f2: function (f1) {
        f1();
        arguments[0]();
    }
}
obj.f2(f1,1,2)
```



# 2、全局变量作用域

当全局变量和局部变量同名时，全局变量不会作用于同名的局部变量的作用域

### 预编译

```js
var a = 10;
function test() {
    console.log(a);
    a=100;
    console.log(this.a);
    var a;
    console.log(a);
}
test();
```
